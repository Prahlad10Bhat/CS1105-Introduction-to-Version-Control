1. mkdir WebProject
   cd WebProject
   git init
   git config --global user.name "Prahlad10Bhat"
   git config --global user.email "prahlad10bhat@gmail.com"
   echo ".DS_Store" >> .gitignore
   echo "node_modules/" >> .gitignore
   git remote add origin https://github.com/Prahlad10Bhat/Git-demo
   echo "This is a sample text file" > sample.txt
   git add sample.txt
   git commit -m "Initial commit"
   git branch -M main 
   git push -u origin main

2. git checkout -b feature-login
   echo "<form>Login Form</form>" > login.html
   git add login.html
   git commit -m "Added basic login form in login.html"
   echo "form { padding: 20px; }" > style.css
   git add style.css
   git commit -m "Added style.css to style login form"
   git push origin feature-login

3. git checkout -b development
   git merge --no-ff feature-login -m "Merged feature-login into development"
   git push origin development

4. git checkout -b hotfix-login-typo
   code login.html
   git add login.html
   git commit -m "Fixed typo in login.html"
   git checkout development
   git merge hotfix-login-typo --ff-only
   git push origin development

5. git checkout development
   git tag -a release-1.0 -m "Release 1.0 - Stable version"
   git push --all origin    
   git push --tags         
   git tag -l
   cd ..
   git clone --branch release-1.0 --depth 1 https://github.com/Prahlad10Bhat/Git-demo WebProject-Release
   cd WebProject-Release
   git status
   git describe --tags
